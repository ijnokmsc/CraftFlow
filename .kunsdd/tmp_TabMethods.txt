    private void LoadTabIcons()
    {
        try
        {
            var tp = Plugin.TextureProvider;
            _equipShared = tp.GetFromGameIcon(new GameIconLookup(62001, false, true));
            _foodShared  = tp.GetFromGameIcon(new GameIconLookup(62008, false, true));
            _favShared   = tp.GetFromGameIcon(new GameIconLookup(62147, false, true));
            _recShared   = tp.GetFromGameIcon(new GameIconLookup(82096, false, true));
        }
        catch { }
    }

    private static ImTextureID GetIconHandle(ISharedImmediateTexture? shared)
    {
        if (shared is null) return default;
        var wrap = shared.GetWrapOrDefault();
        return wrap is not null ? wrap.Handle : default;
    }

    private void DrawTabBar()
    {
        var tabNames = new[] { "装备武器", "食物药品", "收藏清单", "推荐套装" };
        var tabTypes = new[] { TabType.Equipment, TabType.Consumable, TabType.Favorites, TabType.Recommendations };

        for (int i = 0; i < tabNames.Length; i++)
        {
            if (i > 0) ImGui.SameLine();
            bool isSelected = _currentTab == tabTypes[i];
            if (isSelected)
                ImGui.PushStyleColor(ImGuiCol.Button, new Vector4(0.3f, 0.5f, 0.8f, 1.0f));

            var icon = GetIconHandle(i switch { 0 => _equipShared, 1 => _foodShared, 2 => _favShared, _ => _recShared });
            if (icon.Handle != 0)
            {
                ImGui.Image(icon, new Vector2(16, 16));
                ImGui.SameLine();
            }

            if (ImGui.Button($"{tabNames[i]}###Tab_{tabTypes[i]}"))
                _currentTab = tabTypes[i];

            if (isSelected)
                ImGui.PopStyleColor();
        }
        ImGui.Separator();
    }
